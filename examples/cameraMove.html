Cesium Examples - Move Camera

Relevant Cesium Source

    var camera = viewer.scene.camera;
    var height = camera.positionCartographic.height;
    var lat = camera.positionCartographic.latitude + Cesium.Math.toRadians(2.0);
    var lon = camera.positionCartographic.longitude + Cesium.Math.toRadians(20.0);
    camera.flyTo({
        destination: Cesium.Cartesian3.fromRadians(lon, lat, height),
        duration: 1.0
    });
