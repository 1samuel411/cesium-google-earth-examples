Cesium Examples - Move Camera with Timeouts

Relevant Cesium Source

var camera = viewer.camera;
var tiltCamera = function(count) {
  var idealTilt = Cesium.Math.toRadians(80);  // all angles in the API are in degrees.
  var c0 = 0.95;
  camera.tilt = c0 * camera.tilt + (1 - c0) * idealTilt;

  if (count < 60) {
    setTimeout(function() {
      tiltCamera(count + 1);
    }, 33);
  }
}

tiltCamera(0);

Relevant Google Earth Source

var tiltCamera = function(count) {
  var idealTilt = 80;  // all angles in the API are in degrees.
  var c0 = 0.90;
  var la = ge.getView().copyAsLookAt(ge.ALTITUDE_RELATIVE_TO_GROUND);

  var tilt = la.getTilt();
  tilt = c0 * tilt + (1 - c0) * idealTilt;
  la.setTilt(tilt);
  ge.getView().setAbstractView(la);

  if (count < 60) {
    setTimeout(function() {
      tiltCamera(count + 1);
    }, 33);
  }
}

tiltCamera(0);
